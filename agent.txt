import yaml
from pathlib import Path
from models import (
    SessionState, RawTextExtraction, ValidationResult, 
    MissingTechnicalDetail, AcceptanceCriteria, FinalStory
)
from llm_client import StructuredLLMClient

class JiraStatelessPipeline:
    def __init__(self, llm_client: StructuredLLMClient, prompt_file: str = "prompts.yaml"):
        self.llm = llm_client
        self.prompts = self._load_prompts(prompt_file)

    def _load_prompts(self, filepath: str) -> dict:
        path = Path(filepath)
        with open(path, "r", encoding="utf-8") as file:
            return yaml.safe_load(file)

    def run(self, raw_text: str) -> SessionState:
        print("--- Enterprise Raw-Text Jira Pipeline ---")
        
        # Phase 0: Zero-Shot Extraction
        state = self._extract_initial_state(raw_text)

        # Phase 1: JIT Validation
        state = self._resolve_core_triad(state)

        # Phase 2: Always-Dynamic Grooming
        state = self._gather_technical_context(state)

        # Phase 3: AC Synthesis
        state = self._synthesize_ac(state)

        # Phase 4: Final Artifact Generation
        state = self._generate_final_story(state)

        return state

    def _extract_initial_state(self, raw_text: str) -> SessionState:
        print("\n[SYSTEM] Analyzing raw input...")
        sys_prompt = self.prompts['system_personas']['extractor']
        user_prompt = self.prompts['user_prompts']['extract_raw_text'].format(raw_text=raw_text)
        
        extraction = self.llm.query(sys_prompt, user_prompt, RawTextExtraction)
        
        return SessionState(
            what=extraction.what.evidence if extraction.what.identified else None,
            why=extraction.why.evidence if extraction.why.identified else None,
            who=extraction.who.evidence if extraction.who.identified else None,
            ac_evidence=extraction.ac_evidence
        )

    def _resolve_core_triad(self, state: SessionState) -> SessionState:
        sys_prompt = self.prompts['system_personas']['validator']
        max_retries = 3

        for field in ['what', 'why', 'who']:
            retries = 0
            while getattr(state, field) is None:
                if retries >= max_retries:
                    print(f"\n[SYSTEM WARN] Threshold exceeded. Forcing raw input for {field.upper()}.")
                    setattr(state, field, f"Unknown {field}")
                    break

                user_ans = input(f"\nMissing {field.upper()}. Please clarify:\n> ").strip()
                if not user_ans:
                    continue

                user_prompt = self.prompts['user_prompts']['validate_missing'].format(
                    field_name=field.upper(), user_ans=user_ans
                )

                validation = self.llm.query(sys_prompt, user_prompt, ValidationResult)

                if validation.is_valid:
                    setattr(state, field, validation.normalized_value)
                else:
                    retries += 1
                    reason = validation.rejection_reason or "Invalid input."
                    print(f"\n[REJECTED] {reason} (Attempt {retries}/{max_retries})")

        return state

    def _gather_technical_context(self, state: SessionState) -> SessionState:
        print("\n[SYSTEM] Generating dynamic grooming questions...")
        sys_prompt = self.prompts['system_personas']['tech_lead']
        user_prompt = self.prompts['user_prompts']['generate_dynamic_questions'].format(what_statement=state.what)
        
        analysis = self.llm.query(sys_prompt, user_prompt, MissingTechnicalDetail)
        
        if analysis.has_missing_details and analysis.questions:
            questions = analysis.questions[:4]
            for i, q in enumerate(questions, 1):
                ans = input(f"\nTech Detail {i}/{len(questions)}:\n{q}\n(Answer, or type 'skip'):\n> ").strip()
                if ans.lower() not in ["skip", "no", "n/a", ""]:
                    state.tech_notes.append(f"Q: {q} | A: {ans}")
        else:
            print("[SYSTEM] No technical questions needed.")
            
        return state

    def _synthesize_ac(self, state: SessionState) -> SessionState:
        print("\n[SYSTEM] Synthesizing Acceptance Criteria...")
        sys_prompt = self.prompts['system_personas']['agile_coach']
        user_prompt = self.prompts['user_prompts']['generate_ac'].format(
            who=state.who, what=state.what, why=state.why, tech_notes=state.tech_notes,
            ac_evidence=state.ac_evidence or "None"
        )
        
        ac_draft = self.llm.query(sys_prompt, user_prompt, AcceptanceCriteria)
        current_ac = ac_draft.criteria
        
        for attempt in range(3):
            print("\n--- Proposed AC ---")
            for idx, crit in enumerate(current_ac, 1):
                print(f"{idx}. {crit}")
                
            feedback = input(f"\nConfirm AC? (Type 'confirm' or edits. Edits left: {3 - attempt}):\n> ").strip()
            if feedback.lower() == 'confirm':
                break
                
            update_prompt = self.prompts['user_prompts']['update_ac'].format(
                ac_json={"criteria": current_ac}, feedback=feedback
            )
            current_ac = self.llm.query(sys_prompt, update_prompt, AcceptanceCriteria).criteria

        state.ac_list = current_ac
        return state

    def _generate_final_story(self, state: SessionState) -> SessionState:
        print("\n[SYSTEM] Generating final artifact...")
        sys_prompt = self.prompts['system_personas']['story_writer']
        user_prompt = self.prompts['user_prompts']['generate_story'].format(
            who=state.who, what=state.what, why=state.why, agreed_ac=state.ac_list
        )
        
        story_draft = self.llm.query(sys_prompt, user_prompt, FinalStory)
        current_story = story_draft
        
        for attempt in range(3):
            print("\n" + "="*50 + "\nFINAL AGILE JIRA STORY\n" + "="*50)
            print(current_story.user_story)
            print("\nAC:")
            for idx, crit in enumerate(current_story.acceptance_criteria, 1):
                print(f"{idx}. {crit}")
                
            feedback = input(f"\nConfirm Story? (Type 'confirm' or edits. Edits left: {3 - attempt}):\n> ").strip()
            if feedback.lower() == 'confirm':
                break
                
            update_prompt = self.prompts['user_prompts']['update_story'].format(
                story_json=current_story.model_dump_json(), feedback=feedback
            )
            current_story = self.llm.query(sys_prompt, update_prompt, FinalStory)

        state.final_story = current_story.user_story
        state.ac_list = current_story.acceptance_criteria
        return state
