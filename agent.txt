import yaml
import asyncio
from typing import TypedDict, List, Optional
from langgraph.graph import StateGraph, START, END
from llm_client import StructuredLLMClient
from models import (
    SessionState, RawTextExtraction, ValidationResult, 
    MissingTechnicalDetail, TechAnswerValidation, AgileArtifact
)

class GraphState(TypedDict):
    user_input: str
    
    # Core Memory
    who: Optional[str]
    what: Optional[str]
    why: Optional[str]
    ac_evidence: Optional[str]
    pending_questions: List[str]
    tech_notes: List[str]
    
    # Unified Artifact
    final_user_story: Optional[str]
    final_ac_list: List[str]
    
    # Retry Guardrails
    phase1_retries: int
    synthesis_retries: int
    
    # Flow Control Flags
    sys_message: str
    is_complete: bool

class JiraStatelessPipeline:
    def __init__(self, llm_client: StructuredLLMClient):
        self.llm = llm_client
        with open("prompts.yaml", "r") as f:
            self.prompts = yaml.safe_load(f)
            
        self.personas = self.prompts["system_personas"]
        self.templates = self.prompts["user_prompts"]
        self.graph = self._build_graph()

    def run(self, raw_input: str) -> SessionState:
        """
        CLI Entrypoint. Bootstraps the Asyncio event loop.
        """
        return asyncio.run(self._orchestrate_cli_loop(raw_input))

    async def _orchestrate_cli_loop(self, initial_input: str) -> SessionState:
        state: GraphState = {
            "user_input": initial_input,
            "who": None, "what": None, "why": None, "ac_evidence": None,
            "pending_questions": [], "tech_notes": [], 
            "final_user_story": None, "final_ac_list": [],
            "phase1_retries": 0, "synthesis_retries": 0,
            "sys_message": "", "is_complete": False
        }

        while not state["is_complete"]:
            state = await self.graph.ainvoke(state)
            
            if state["is_complete"]:
                print(f"\n[AGENT] {state['sys_message']}")
                break
                
            print(f"\n[AGENT] {state['sys_message']}")
            user_val = input("\n[USER] > ").strip()
            state["user_input"] = user_val if user_val else "[INIT]"

        final_session = SessionState(
            what=state["what"], why=state["why"], who=state["who"],
            ac_evidence=state["ac_evidence"], tech_notes=state["tech_notes"],
            final_user_story=state["final_user_story"],
            final_ac_list=state["final_ac_list"]
        )
        return final_session

    def _build_graph(self):
        workflow = StateGraph(GraphState)

        workflow.add_node("phase0_extractor", self._node_extractor)
        workflow.add_node("phase1_validator", self._node_validator)
        workflow.add_node("phase2_tech", self._node_tech_grooming)
        workflow.add_node("phase3_unified", self._node_unified_synthesis)

        workflow.add_conditional_edges(START, self._route_from_start)
        workflow.add_conditional_edges("phase0_extractor", self._route_general)
        workflow.add_conditional_edges("phase1_validator", self._route_general)
        workflow.add_conditional_edges("phase2_tech", self._route_general)
        workflow.add_conditional_edges("phase3_unified", self._route_general)

        return workflow.compile()

    # ==========================================
    # ROUTING LOGIC
    # ==========================================
    def _route_from_start(self, state: GraphState) -> str:
        if not state["what"] and not state["who"] and not state["why"]:
            return "phase0_extractor"
        return self._route_general(state)

    def _route_general(self, state: GraphState) -> str:
        if state["is_complete"]:
            return END
        if not state["what"] or not state["who"] or not state["why"]:
            return "phase1_validator"
        if getattr(state, "_needs_tech_init", True) or state["pending_questions"]:
            return "phase2_tech"
        if not getattr(state, "_artifact_confirmed", False):
            return "phase3_unified"
        return END

    # ==========================================
    # GRAPH NODES
    # ==========================================
    async def _node_extractor(self, state: GraphState) -> GraphState:
        prompt = self.templates["extract_raw_text"].format(raw_text=state["user_input"])
        ext: RawTextExtraction = await self.llm.query(self.personas["extractor"], prompt, RawTextExtraction)
        
        state["what"] = ext.what.evidence if ext.what.identified else None
        state["who"] = ext.who.evidence if ext.who.identified else None
        state["why"] = ext.why.evidence if ext.why.identified else None
        state["ac_evidence"] = ext.ac_evidence
        state["user_input"] = "[INIT]"
        return state

    async def _node_validator(self, state: GraphState) -> GraphState:
        missing = [f for f in ["what", "why", "who"] if not state[f]]
        if not missing:
            return state

        field = missing[0]
        
        # Hard Abort execution condition
        if state["phase1_retries"] >= 3:
            state["sys_message"] = f"ABORT: Failed to resolve '{field}' after 3 attempts. Pipeline Terminated."
            state["is_complete"] = True
            return state

        if state["user_input"] == "[INIT]":
            state["sys_message"] = f"Missing field '{field.upper()}'. Please provide details."
            return state

        prompt = self.templates["validate_missing"].format(field_name=field, user_ans=state["user_input"])
        val: ValidationResult = await self.llm.query(self.personas["validator"], prompt, ValidationResult)
        
        if val.is_valid:
            state[field] = val.normalized_value
            state["phase1_retries"] = 0
            state["user_input"] = "[INIT]"
        else:
            state["phase1_retries"] += 1
            state["sys_message"] = f"Rejected ({state['phase1_retries']}/3): {val.rejection_reason}. Try again."
            
        return state

    async def _node_tech_grooming(self, state: GraphState) -> GraphState:
        if getattr(state, "_needs_tech_init", True):
            prompt = self.templates["generate_dynamic_questions"].format(what_statement=state["what"])
            tech: MissingTechnicalDetail = await self.llm.query(self.personas["tech_lead"], prompt, MissingTechnicalDetail)
            state["pending_questions"] = tech.questions
            state["_needs_tech_init"] = False
            state["user_input"] = "[INIT]"
            
            if not state["pending_questions"]:
                return state

        if state["user_input"] != "[INIT]" and state["pending_questions"]:
            eval_prompt = self.templates["evaluate_tech_answer"].format(
                user_answer=state['user_input'], question=state['pending_questions'][0]
            )
            eval_res: TechAnswerValidation = await self.llm.query(
                self.personas["tech_evaluator"], eval_prompt, TechAnswerValidation
            )
            
            if eval_res.is_valid:
                state["tech_notes"].append(f"Q: {state['pending_questions'][0]} | A: {state['user_input']}")
            
            state["pending_questions"].pop(0)

        if state["pending_questions"]:
            state["sys_message"] = f"Tech Question: {state['pending_questions'][0]}"
        else:
            state["user_input"] = "[INIT]"
            
        return state

    async def _node_unified_synthesis(self, state: GraphState) -> GraphState:
        # Force Commit execution condition
        if state["synthesis_retries"] >= 3:
            state["_artifact_confirmed"] = True
            state["is_complete"] = True
            state["sys_message"] = "Feedback limit reached (3/3). Force committing the artifact to preserve pipeline stability."
            return state

        if not state["final_user_story"]:
            prompt = self.templates["generate_agile_artifact"].format(
                who=state["who"], what=state["what"], why=state["why"],
                tech_notes="\n".join(state["tech_notes"]), ac_evidence=state["ac_evidence"] or "None"
            )
            artifact: AgileArtifact = await self.llm.query(self.personas["agile_coach"], prompt, AgileArtifact)
            
            state["final_user_story"] = artifact.user_story
            state["final_ac_list"] = artifact.acceptance_criteria
            
            ac_text = "\n".join([f"{i+1}. {ac}" for i, ac in enumerate(state["final_ac_list"])])
            state["sys_message"] = f"Draft Agile Artifact:\n\n{state['final_user_story']}\n\nAcceptance Criteria:\n{ac_text}\n\nType 'confirm' to finalize or provide global feedback."
            return state

        if state["user_input"].strip().lower() == "confirm":
            state["_artifact_confirmed"] = True
            state["is_complete"] = True
            state["sys_message"] = "Pipeline Execution Completed Successfully. Artifact locked."
            return state

        # Handling Global Feedback
        prompt = self.templates["update_agile_artifact"].format(
            current_story=state["final_user_story"],
            current_ac="\n".join(state["final_ac_list"]),
            feedback=state["user_input"]
        )
        updated_artifact: AgileArtifact = await self.llm.query(self.personas["agile_coach"], prompt, AgileArtifact)
        
        state["final_user_story"] = updated_artifact.user_story
        state["final_ac_list"] = updated_artifact.acceptance_criteria
        state["synthesis_retries"] += 1
        
        ac_text = "\n".join([f"{i+1}. {ac}" for i, ac in enumerate(state["final_ac_list"])])
        state["sys_message"] = f"Artifact Updated (Attempt {state['synthesis_retries']}/3):\n\n{state['final_user_story']}\n\nAcceptance Criteria:\n{ac_text}\n\nType 'confirm' to finalize or provide feedback."
        return state
